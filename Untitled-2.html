let currentSlide = 0; const carousel = document.getElementById('carousel'); const totalSlides = carousel.children.length; function moveCarousel(direction) { currentSlide += direction; if (currentSlide < 0) { currentSlide = totalSlides - 1; } else if (currentSlide >= totalSlides) { currentSlide = 0; } carousel.style.transform = `translateX(-${currentSlide * 100}%)`; }
